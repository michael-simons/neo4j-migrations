package ac.simons.neo4j.migrations.core.catalog;

import java.util.Objects;

/**
 * Value holder for a names value. Default and generated names are different classes on purpose, even though they could
 * be the same one, with the generator on the interface.
 *
 * @author Michael J. Simons
 * @since 1.7.0
 */
abstract non-sealed class AbstractName implements Name {

	/**
	 * Value of this name, might be {@literal null} or blank.
	 */
	private final String value;

	AbstractName(String value) {
		this.value = value;
	}

	@Override
	public String getValue() {
		return value;
	}

	@Override public boolean equals(Object o) {
		if (this == o) {
			return true;
		}
		if (o == null || getClass() != o.getClass()) {
			return false;
		}
		AbstractName that = (AbstractName) o;
		return Objects.equals(value, that.value);
	}

	@Override
	public int hashCode() {
		return Objects.hash(value);
	}

	@Override
	public String toString() {
		return value;
	}
}
